    './component.js',
    'text!./component.html',
    'text!./component.css',
    '../check/ui.js',
    'pool/component-validation/src/validation'
],function(
    CheckGroup,
    html,
    css
){
    /**
     * CheckGroup UI组件
     *
     * @class   module:pool/component-check/src/check_group/ui.CheckGroupUI

     */

        name     : 'ux-check-group',
        css      : css,
        template : html
    });
});
